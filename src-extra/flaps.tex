\newcommand{\hsep}[3]{%
  \fontsize{210pt}{0}
  \begin{tikzpicture}
    \draw [fill=#2] (0,0) rectangle (4.6cm, 5.6cm);
    \draw [color=lightgray] (0,2.8cm) -- (4.6cm, 2.8cm);
    \node [color=#3,anchor=south east, overlay] at (6.3,-2.3) {#1};
  \end{tikzpicture}
}
